# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseBetween(self, head: Optional[ListNode], left: int, right: int) -> Optional[ListNode]:
        if left==right:
            return head
        dummy=ListNode(0)
        dummy.next=head
        prev=dummy

        for i in range(left-1):
            prev=prev.next
        a=prev.next
        b=a.next

        for i in range(right-left):
            temp=b.next
            b.next=a
            a=b
            b=temp
        prev.next.next=b
        prev.next=a
        return dummy.next
