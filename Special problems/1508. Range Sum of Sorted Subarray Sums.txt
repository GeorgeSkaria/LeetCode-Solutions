class Solution:
    def rangeSum(self, nums: List[int], n: int, left: int, right: int) -> int:
        new=[]
        for i in range(n):
            new.append(nums[i])
            s=nums[i]
            for j in range(i+1,n):
                s+=nums[j]
                new.append(s)
        new.sort()
        return sum(new[left-1:right])%(10**9+7)
