class Solution:
    def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
        # Cycle detection in a directed graph
        d=defaultdict(list)
        for i,j in prerequisites:
            d[j].append(i)
        
        state=[0]*numCourses
        def has_cycle(i):
            if state[i]==1:
                return True
            if state[i]==2:
                return False

            state[i]=1
            for j in d[i]:
                if has_cycle(j):
                    return True
            state[i]=2
            return False

        for i in range(numCourses):
            if has_cycle(i):
                return False        
        return True
